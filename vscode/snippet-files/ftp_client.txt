Name : FTP_Client
Trigger : client/ftp
Imports : wso2/ftp

ftp:ClientEndpointConfig ftpConfig = {
    protocol: ftp:FTP,
    host: "${ftp_host}",
    port: ${ftp_port},
    secureSocket: {
        basicAuth: {
            username: "${ftp_username}",
            password: "${ftp_password}"
        }
    }
};

ftp:Client ftpClient = new(ftpConfig);
